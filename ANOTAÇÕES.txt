#INSTALAR GLOBALMENTE O ANGULAR-CLI 2
npm install -g angular-cli

#OU AS NOVAS VERSÕES
npm install -g @angular/cli

#CRIAR PROJETO
ng new nome-do-projeto

#GERAR UM COMPONENTE
ng g c NOME_COMPONENTE

#ONDE, g = generate, c = component

#CRIAR DENTRO DE UMA PASTA
ng g c pasta/nome-do-componente --flat

#CRIAR DENTRO DE UMA PASTA (SUB-PASTA)
ng g c pasta/nome-do-componente

#COMPONENTE O JEITO MAIS CERTO
ng g c contatos/busca/contatoBusca --flat --module contatos/contatos.module

# Component
ng g c cursos

# Service
ng g s cursos

# Module
ng g m cursos

# Directive
ng g d destaque

# Pipe
ng g p telefone
